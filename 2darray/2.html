    void sumMatrix(int arr1[2][2], int arr2[2][2], int result[2][2]){
        int i, j;
        for (i = 0; i < 2; i++)
        {
        for (j = 0; j < 2; j++)
            {
                result[i][j] = arr1[i][j] + arr2[i][j];
            }
        }
    }

    void printArr(int arr[2][2]){
        int i, j;
        for (i = 0; i < 2; i++)
        {
        for (j = 0; j < 2; j++)
        {
            printf("%d\t", arr[i][j]);
        }
        printf("\n");
        }

    }
