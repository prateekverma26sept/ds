    int binarySearch(char arr[], int l, int u, char key) {
        int mid = l + (u - l) / 2;
        if (l > u) 
        {
            return -1;
        }
        if (arr[mid] == key)
        {
            return mid;
        }
        else if (arr[mid] > key){
            u = mid-1;
        }
        else{
            l = mid+1;
        }
        
        return binarySearch(arr, l, u, key); 
    }
    
    void main() {
        char arr[20], key;
        int result;
        clrscr();
        printf("Enter any String\n");
        gets(arr);
    
        printf("Enter Value to be searched\n");
        scanf("%c", &key);
        result = binarySearch(arr, 0, strlen(arr)-1, key);
    
        if (result == -1) {
        printf("Element %c not found in the array.\n", key);
        } else {
        printf("Element %c found at position %d.\n", key, result+1);
        }
    
        delay(5000);
    }
