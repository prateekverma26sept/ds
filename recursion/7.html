    int sumOfDigits(int num)
    {
        if (num == 0)
        {
            return 0;
        }
        else
        {
            return num % 10 + sumOfDigits(num / 10);
        }
    }
    
    void main()
    {
        int number, sum;
        clrscr();
        printf("Enter a number: ");
        scanf("%d", &number);
    
        sum = sumOfDigits(number);

    }
