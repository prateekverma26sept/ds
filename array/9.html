    void sort(int arr[], int size){
        int i , j;
        for (i = 0; i < size; i++){
            for (j = i+1; j < size; j++){
                if (arr[i] > arr[j]){
                    int temp = arr[i];
                    arr[i] = arr[j];
                    arr[j] = temp;
                }
            }
        }
    }
