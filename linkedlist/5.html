#include <stdio.h>
    #include <alloc.h>

    struct Node
    {
        int data;
        struct Node *next;
    };
    
    int isEmpty(struct Node *start)
    {
        if (start == NULL)
        {
            return 1;
        }
        return 0;
    }
    
    int isFull(struct Node *ptr)
    {
        if (ptr == NULL)
        {
            return 1;
        }
        return 0;
    }
    
    int insAtBeg(struct Node **start, struct Node **end, int value, int *count)
    {
        struct Node *nn = (struct Node *)malloc(sizeof(struct Node));
        if (nn != NULL)
        {
            nn->data = value;
            nn->next = *start;
            (*start) = nn;
            if ((*end) == NULL)
            {
                (*end) = nn;
            }
            (*end)->next = (*start);
            (*count)++;
            return 1;
        }
        return 0;
    }
    
    int insAtEnd(struct Node **start, struct Node **end, int value, int *count)
    {
        if (isEmpty(*start))
        {
            return insAtBeg(start, end, value, count);
        }
        else
        {
            struct Node *nn = (struct Node *)malloc(sizeof(struct Node));
            if (nn != NULL)
            {
                nn->data = value;
                nn->next = *start;
                (*end)->next = nn;
                (*end) = nn;
                (*count)++;
                return 1;
            }
        }
        return 0;
    }
    
    int insAtPos(struct Node **start, struct Node **end, int value, int position, int *count)
    {
        if (position == 1)
        {
            return insAtBeg(start, end, value, count);
        }
        else if (position == (*count) + 1)
        {
            return insAtEnd(start, end, value, count);
        }
        else if (position <= 0 || position > (*count) + 1)
        {
            printf("Invalid Position Number\n");
            return 0;
        }
        else
        {
            int i;
            struct Node *search;
            struct Node *nn = (struct Node *)malloc(sizeof(struct Node));
            search = *start;
            for (i = 1; i < position - 1; i++)
            {
                search = search->next;
            }
            if (nn != NULL)
            {
                nn->data = value;
                nn->next = search->next;
                search->next = nn;
                (*count)++;
                return 1;
            }
        }
        return 0;
    }
    
    int delfromBeg(struct Node **start, struct Node **end, int *count)
    {
        struct Node *temp;
        if (*start == NULL)
        {
            printf("List is empty. Cannot delete.\n");
            return 0;
        }
    
        temp = *start;
        *start = (*start)->next;
    
        if (*start == temp)
        {
            *start = NULL;
            *end = NULL;
        }
        else{
            (*end)->next = *start;
        }
        free(temp);
        (*count)--;
        temp = NULL;
        return 1;
    }
    
    int delfromEnd(struct Node **start, struct Node **end, int *count)
    {
        struct Node *temp, *search;
        if (*start == NULL)
        {
            printf("List is empty. Cannot delete.\n");
            return 0;
        }
    
        temp = *end;
        search = *start;
    
        if ((*start)->next == *start)
        {
            *start = NULL;
            *end = NULL;
        }
        else
        {
            while (search->next != (*end))
            {
                search = search->next;
            }
            *end = search;
            (*end)->next = *start;
        }
    
        free(temp);
        (*count)--;
        temp = NULL;
        return 1;
    }
    
    int delfromPos(struct Node **start, struct Node **end, int position, int *count)
    {
        if (position == 1)
        {
        return delfromBeg(start, end, count);
        }
        else if (position == (*count) + 1)
        {
        return delfromEnd(start, end, count);
        }
        else if (position <= 0 || position > (*count) + 1)
        {
        printf("Invalid Position Number\n");
        return 0;
        }
        else
        {
        int i;
        struct Node *search, *temp;
        search = *start;
        for (i = 1; i < position - 1; i++)
        {
            search = search->next;
        }
        temp = search->next;
        search->next = temp->next;
        free(temp);
        temp = NULL;
        search = NULL;
        (*count)--;
        return 1;
        }
    }
    
    void display(struct Node *start)
    {
        struct Node *ptr = start;
        if (ptr == NULL)
        {
            printf("List is Empty\n");
            return;
        }
    
        do 
        {
            printf("%d\n", ptr->data);
            ptr = ptr->next;
        }while(ptr != start);
    }
    
    int *LinearSearch(struct Node *start, int key)
    {
        struct Node *ptr = start;
        if (ptr == NULL)
        {
            printf("List is Empty\n");
            return NULL;
        }
    
        do
        {
            if (ptr->data == key)
            {
                return &(ptr->data);
            }
            ptr = ptr->next;
        }while(ptr != start);
        
        return NULL;
    }
    
    void main()
    {
        struct Node *start = NULL, *end = NULL;
        int count = 0, val, pos;

            printf("Press 1: Insert\nPress 2: Delete\nPress 3: Display List\nPress 4: Search\nPress 5: Exit\n\n");

                        printf("Insertion Successfull\n");

                    if (delfromBeg(&start, &end, &count))
                    {
                        printf("Deletion Successfull\n");
                    }
                }

                    printf("Enter Position\n");

                printf("Enter Value to be Searched\n");

            {

            }
