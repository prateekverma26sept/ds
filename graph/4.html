#include #include // Structure for a node in the adjacency list struct Node { int data; struct Node* next; }; // Structure for the adjacency list struct Graph { int Vertix; // Number of vertices struct Node** array; }; // Function to create a new node struct Node* createNode(int data) { struct Node* newNode = (struct Node*)malloc(sizeof(struct Node)); newNode->data = data; newNode->next = NULL; return newNode; } // Function to create a graph with 'V' vertices struct Graph* createGraph(int V) { struct Graph* graph = (struct Graph*)malloc(sizeof(struct Graph)); graph->Vertix = V;//set value graph->array = (struct Node**)malloc(V * sizeof(struct Node*)); for (int i = 0; i < V; i++) { // each element in the array is initialized to NULLs graph->array[i] = NULL; } return graph; } // Function to add an edge to the graph void addEdge(struct Graph* graph, int src, int dest) { struct Node* newNode = createNode(dest); newNode->next = graph->array[src]; graph->array[src] = newNode; // For undirected graph, add an edge from dest to src as well newNode = createNode(src); newNode->next = graph->array[dest]; graph->array[dest] = newNode; } // Function to perform DFS traversal of the graph void DFSUtil(struct Graph* graph, int vertex, int visited[]) { visited[vertex] = 1; printf("%d ", vertex); struct Node* currentNode = graph->array[vertex]; while (currentNode != NULL) { int adjVertex = currentNode->data; if (!visited[adjVertex]) { DFSUtil(graph, adjVertex, visited); } currentNode = currentNode->next; } } // Function to initiate DFS traversal void DFS(struct Graph* graph, int startVertex) { int* visited = (int*)malloc(graph->Vertix * sizeof(int)); for (int i = 0; i < graph->Vertix; i++) { visited[i] = 0; } DFSUtil(graph, startVertex, visited); free(visited); } int main() { int V = 6; // Number of vertices in the graph struct Graph* graph = createGraph(V); addEdge(graph, 0, 1); addEdge(graph, 0, 2); addEdge(graph, 1, 3); addEdge(graph, 2, 4); addEdge(graph, 3, 5); printf("DFS traversal starting from vertex 0: "); DFS(graph, 0); return 0; } // OUTPUT :- // DFS traversal starting from vertex 0: 0 2 4 1 3 5
